## JS Code

```javascript
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number[]}
 */
var twoSum = function (nums, target) {
  const lib = {}

  return nums.reduce((acc, num, idx) => {
    const needNum = target - num

    if (!isNaN(lib[needNum]) && !acc.length) {
      return [lib[needNum], idx]
    }

    lib[num] = idx
    return acc
  }, [])
}
```
