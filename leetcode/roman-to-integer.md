```javascript
/**
 * @param {string} s
 * @return {number}
 */
var romanToInt = function(s) {
  let answer = 0
  let index = 0
  const romans = [
    [1000, 'M'],
    [900, 'CM'],
    [500, 'D'],
    [400, 'CD'],
    [100, 'C'],
    [90, 'XC'],
    [50, 'L'],
    [40, 'XL'],
    [10, 'X'],
    [9, 'IX'],
    [5, 'V'],
    [4, 'IV'],
    [1, 'I'],
  ]
  
  for(const [value, symbol] of romans) {
      while (s.indexOf(symbol,index) === index) {
          answer += value
          index += symbol.length
      }
  }
    
    return answer
};
```
